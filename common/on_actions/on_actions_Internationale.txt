on_actions = {
	#ROOT is winner #FROM is loser
	on_annex = {
		effect = {
			# Reset the INT alias
			if = {
				limit = { FROM = { tag = INT } }
				FRA = { save_global_event_target_as = KR_internationale_leader }
				every_country = {
					limit = { is_in_tech_sharing_group = phalanstere_research }
					remove_from_tech_sharing_group = phalanstere_research
				}
			}
			else = {
				FROM = { remove_from_tech_sharing_group = phalanstere_research }
			}
		}
	}
	# ROOT is capitulated country, FROM is winner
	on_capitulation_immediate = {
		effect = {
			# France capitulates
			if = {
				limit = {
					tag = FRA
					NOT = { has_global_flag = FRA_capitulated }
				}
				country_event = { id = capt.40 days = 1 }
			}

			# Move the Internationale leader
			if = {
				limit = { tag = INT }
				FRA_transfer_internationale_leader = yes
			}
		}
	}

	#FROM is war target
	on_declare_war = {
		effect = {
			# CoF events
			if = {
				limit = {
					ROOT = { tag = FRA }
					FROM = {
						tag = BEL
						is_subject = no
						is_in_faction = no
					}
				}
				FROM = {
					country_event = {
						id = comfrance.166
						days = 2
					}
				}
			}
			if = {
				limit = {
					ROOT = { tag = FRA }
					FROM = {
						tag = HOL
						is_subject = no
						is_in_faction = no
					}
				}
				FROM = {
					country_event = {
						id = comfrance.167
						days = 2
					}
				}
			}
		}
	}
	#ROOT is new controller #FROM is old controller #FROM.FROM is state ID
	on_state_control_changed = {
		effect = {
			### A New 1812/1328 (event)
			if = {
				limit = {
					FROM = { tag = FRA }
					FROM.FROM = {
						OR = {
							state = 24
							state = 785
						}
					}
				}
				FROM = {
					country_event = {
						id = comfrance.23
						hours = 6
					}
				}
			}
		}
	}
	#ROOT = attacking side
	#FROM = defending side
	#fired when two countries end up at war with each other (on_war is fired when a country goes to war against anyone and is not fired again when it enters war against another country unless it went to peace first)
	on_war_relation_added = {
		effect = {
			if = {
				limit = {
					is_in_tech_sharing_group = phalanstere_research
					INT = { is_ally_with = FROM }
				}
				remove_from_tech_sharing_group = phalanstere_research
			}
			else_if = {
				limit = {
					FROM = { is_in_tech_sharing_group = phalanstere_research }
					is_ally_with = INT
				}
				FROM = { remove_from_tech_sharing_group = phalanstere_research }
			}
		}
	}
}
